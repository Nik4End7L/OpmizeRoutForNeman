`delivery_bot/
│
├── .env                    # Скрытый файл с токеном Telegram бота и API ключами
├── requirements.txt        # Список зависимостей (aiogram, requests, ortools)
├── run.py                  # Главный файл запуска скрипта
│
├── config.py               # Загрузка настроек из файла .env
│
├── routing/                # Модуль логики построения маршрутов
│   ├── __init__.py
│   ├── api_client.py       # Запросы к OSRM/Яндекс для получения времени и дистанции между адресами
│   └── optimizer.py        # Алгоритм сортировки 10 адресов в оптимальном порядке
│
├── bot/                    # Модуль Telegram-бота
│   ├── __init__.py
│   ├── handlers.py         # Обработчики команд (/start, нажатие кнопок "Доставил")
│   ├── keyboards.py        # Кнопки для курьера (например, "Следующий адрес", "Показать на карте")
│   └── states.py           # FSM (машина состояний) для управления процессом доставки
│
└── database/               # Модуль работы с БД
    ├── __init__.py
    └── db.py               # Логика сохранения текущего заказа (SQLite)`
